{-# OPTIONS --cubical --safe #-}
module Fixed where

open import Cubical.Foundations.Prelude hiding (_[_↦_])
open import Cubical.Foundations.Function
open import Cubical.Data.Nat


-- The idea: Fx A is the type A extended with arbitrary fixed points.

-- ∙ Thus: ⇓ a : Fx A for any a ∈ A and fix f : Fx A for any f ∈ Fx (A → A).

-- ∙ A good intuition is to think of Fx A as the Haskell version of the Agda
--   type A. Thus, for example, ⊥ : Fx A for all types A, and Fx (List A) may
--   contain infinite lists.

-- ∙ We also postulate that elements of Fx (A → B) are the same as functions A → Fx B.
--   (This follows the way we actually define functions in Haskell!)

-- ∙ To make all this work, we postulate lazy substitutions:
--    e(x) [ x → y ] : Fx A where e(x) : Fx A has a free variable x : X, and y : Fx X.
--   If y ≡ ⇓ y', this is just a substitution of y' for x in e. Otherwise, we may have
--   to evaluate y further before applying it. We also postulate:
--    e [ x → y ] ≡ e where e : Fx A does not depend on x
--   which gives this substitution its 'lazy' character. Thus, we define:
--    fix f ≡ f x [ x ↦ fix f ] for any f : A → Fx A.

-- ∙ All of this is done in a higher inductive type!


variable
  ℓ : Level

syntax  lam (λ a → e) = λₓ a →ₓ e
infix 5 lam
syntax  sub (λ x → e) y = e [ x ↦ y ]
infix 6 sub
infix 7 ⇓_


data Fx {ℓ} : Type ℓ → Type (ℓ-suc ℓ) where

   -- lazy substitutions, written e [ x ↦ y ] = sub (λ x → e) y
   --  Note that y : Fx X, while the hole x in e is of type X. As we establish below, if x ≡ ⇓ z then
   --  this is the same as just substituting z for x. Otherwise, we may have to 'evaluate' z further.
   sub : ∀ {X A : Type ℓ} → (X → Fx A) → Fx X → Fx A

   -- everything in A is also in Fx A
   ⇓_ : ∀ {A : Type ℓ} → A → Fx A

   -- fixed points exist for every type! Morally, fix has type Fx (A → A) → Fx A, but as we
   --  establish below, the types Fx (A → A) and (A → Fx A) are equivalent.
   fix : ∀ {A : Type ℓ} → (A → Fx A) → Fx A
   unfold : ∀ {A : Type ℓ} (f : A → Fx A) → fix f ≡ f x [ x ↦ fix f ] -- lazy substitution!

   -- the type Fx (A → B) is exactly the type of functions (A → Fx B)
   ap  : ∀ {A B : Type ℓ} → Fx (A → B) → A → Fx B
   lam : ∀ {A B : Type ℓ} → (A → Fx B) → Fx (A → B) -- Notation: λₓ a →ₓ e = lam (λ a → e)
   ap-lam : ∀ {A B : Type ℓ} (f : A → Fx B)   → ap (λₓ a →ₓ f a) ≡ f
   lam-ap : ∀ {A B : Type ℓ} (f : Fx (A → B)) → (λₓ a →ₓ ap f a) ≡ f

   -- lazy substitions behave as we expect on arbitrary terms
   sub-⇓ : ∀ {X A : Type ℓ} (a : X → Fx A) (z : X) → a(x) [ x ↦ ⇓ z ] ≡ a(z)
   sub-const : ∀ {X A : Type ℓ} (a : Fx A) (z : Fx X) → a [ _ ↦ z ] ≡ a
   -- (this ↓ one is very odd -- most of these equalities are pretty ad hoc at the moment)
   sub-ap  : ∀ {X B A : Type ℓ} (a : X → Fx A) (b : B) (y : Fx (B → X)) → a(x) [ x ↦ ap y b ] ≡ a(x b) [ x ↦ y ]

   -- lazy substitions compute on constructors
   lam-sub : ∀ {X A B : Type ℓ} (f : X → (A → Fx B)) (y : Fx X) → (λₓ a →ₓ f(x) a) [ x ↦ y ] ≡ λₓ a →ₓ f(x) a [ x ↦ y ]
   ap-sub  : ∀ {X A B : Type ℓ} (f : X → Fx (A → B)) (a : A) (y : Fx X) → ap (f(x)) a [ x ↦ y ] ≡ ap (f(x) [ x ↦ y ]) a
   fix-sub : ∀ {X A : Type ℓ} (f : X → A → Fx A) (y : Fx X) → fix (f(x)) [ x ↦ y ] ≡ fix (λ a → f(x) a [ x ↦ y ])
   ⇓-sub   : ∀ {X : Type ℓ} (y : Fx X) → (⇓ x) [ x ↦ y ] ≡ y
   sub-sub : ∀ {X Y A : Type ℓ} (a : Y → Fx A) (b : X → Fx Y) (z : Fx X) → a(x) [ x ↦ b(y) ] [ y ↦ z ] ≡ a(x) [ x ↦ b(y) [ y ↦ z ] ]
   
   -- proofs are irrelevant
   trunc : ∀ {A : Type ℓ} {x y : Fx A} (p q : x ≡ y) → p ≡ q


-- Fx A is a category

idₓ : ∀ {A : Type ℓ} → A → Fx A
idₓ x = ⇓ x

_∘ₓ_ : ∀ {A B C : Type ℓ} → (B → Fx C) → (A → Fx B) → (A → Fx C)
(g ∘ₓ f) a = g b [ b ↦ f a ]

idₓˡ : ∀ {A B : Type ℓ} (f : A → Fx B) → idₓ ∘ₓ f ≡ f
idₓˡ f i a = ⇓-sub (f a) i

idₓʳ : ∀ {A B : Type ℓ} (f : A → Fx B) → f ∘ₓ idₓ ≡ f
idₓʳ f i a = sub-⇓ f a i

∘ₓ-assoc : ∀ {A B C D : Type ℓ} (h : C → Fx D) (g : B → Fx C) (f : A → Fx B) → (h ∘ₓ g) ∘ₓ f ≡ h ∘ₓ (g ∘ₓ f)
∘ₓ-assoc h g f i a = sub-sub h g (f a) i


-- Fx A is always inhabited!
⊥ : ∀ {A : Type ℓ} → Fx A
⊥ = fix idₓ


-- a value x : Fx A converges if there is some a : A such that x ≡ ⇓ a
converges : ∀ {A : Type ℓ} (a : Fx A) → Type (ℓ-suc ℓ)
converges {ℓ} {A} x = Σ A (λ a → x ≡ ⇓ a)

-- a function is total if it converges on every input
total : ∀ {A B : Type ℓ} (f : A → Fx B) → Type (ℓ-suc ℓ)
total f = ∀ a → converges (f a)


-- a version of fix that's easier to use with functions

fixf : ∀ {A B : Type ℓ} → ((A → B) → A → Fx B) → Fx (A → B)
fixf f = fix (λ r → λₓ a →ₓ f r a)

unfoldf : ∀ {A B : Type ℓ} (f : (A → B) → A → Fx B) (a : A) → ap (fixf f) a ≡ f x a [ x ↦ fixf f ]
unfoldf f a = 
   ap (fixf f) a                         ≡⟨ (λ i → ap (unfold (λ r → λₓ a →ₓ f r a) i) a) ⟩
   ap ((λₓ a →ₓ f z a) [ z ↦ fixf f ]) a ≡⟨ (λ i → ap (lam-sub (λ z a → f z a) (fixf f) i) a) ⟩
   ap ( λₓ a →ₓ f z a  [ z ↦ fixf f ]) a ≡⟨ (λ i → (ap-lam (λ a → f z a [ z ↦ fixf f ]) i) a) ⟩
   f z a [ z ↦ fixf f ]                  ∎


private

  -- an example of using fix to compute the nth triangle number

  triangle : ℕ → ℕ
  triangle zero = zero
  triangle (suc n) = suc n + triangle n


  tri' : (ℕ → ℕ) → ℕ → Fx ℕ
  tri' _ zero = ⇓ zero
  tri' r (suc n) = ⇓ (suc n + r n)

  tri : ℕ → Fx ℕ
  tri = ap (fixf tri')


  tri-zero : tri zero ≡ ⇓ zero
  tri-zero = unfoldf tri' zero ∙ sub-const (⇓ zero) (fixf tri')

  tri-triangle : ∀ n → tri n ≡ ⇓ triangle n
  tri-triangle zero = tri-zero
  tri-triangle (suc n) =
    tri (suc n)                          ≡⟨ unfoldf tri' (suc n) ⟩
    ⇓ (suc n + r n) [ r ↦ fixf tri' ]   ≡⟨ sym (sub-ap (λ x → ⇓ (suc n + x)) n (fixf tri')) ⟩
    ⇓ (suc n + x) [ x ↦ tri n ]         ≡⟨ (λ i → ⇓ (suc n + x) [ x ↦ tri-triangle n i ]) ⟩
    ⇓ (suc n + x) [ x ↦ ⇓ triangle n ]  ≡⟨ sub-⇓ (λ x → ⇓ (suc n + x)) (triangle n) ⟩
    ⇓ (suc n + triangle n)               ∎
  
  tri-total : total tri
  tri-total n = triangle n , tri-triangle n
